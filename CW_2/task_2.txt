#include <iostream>
using namespace std;

int main()
{
    int m,n;
    cout<<"Input size of matrix (nxn)"<<endl;
    cin>>m;
    n=m;
    int** matrix = new int* [n];
    for (int i = 0; i<n; i++)
    {
        matrix[i] = new int[m];
        for (int j = 0; j<m; j++)
        {
            cin>>matrix[i][j];
        }
    }
    for (int i = 0; i<n; i++)
    {
        for (int j = 0; j<n; j++)
        {
            cout<<matrix[i][j]<<'\t';
        }
        cout<<endl;
    }
    for (int i = 0; i<n; i++)
    {
        delete[] matrix[i];
    }
    return 0;
}
